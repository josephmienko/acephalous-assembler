Machine Info:

  machine.os_family     "Ubuntu" | "Debian" | "Home Assistant OS"
  machine.os_version    version string (e.g., "24.04 LTS")
  machine.network.mode  "dhcp" | "static"
  machine.network.ip_address  IP address or null

Bootstrap Config:
  bootstrap.ssh_user    "ubuntu" | "debian" | "homeassistant"
  bootstrap.ssh_port    22 (integer)
  bootstrap.status      "not_yet_installed"
  bootstrap.marker_supported  true (boolean)
  bootstrap.marker_path "/var/lib/acephalous-assembler/bootstrap-complete"
